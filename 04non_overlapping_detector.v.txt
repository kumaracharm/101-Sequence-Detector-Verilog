module n_o_s_101 (
    input clk,
    input rst,
    input x,
    output out
);

reg [1:0] state;

parameter S0 = 2'b00;
parameter S1 = 2'b01;
parameter S2 = 2'b10;

always @(posedge clk)
begin
    if(rst)
        state <= S0;
    else
        case(state)
            S0: state <= (x) ? S1 : S0;
            S1: state <= (x) ? S1 : S2;
            S2: state <= S0;
        endcase
end

assign out = (state == S2 && x);

endmodule
